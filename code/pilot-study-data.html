Assign taxonomy using Silva 138.1

tax <- assignTaxonomy(st, "~/tax/silva_nr99_v138.1_train_set.fa.gz", multithread=TRUE)
taxsp <- addSpecies(tax, "~/tax/silva_species_assignment_v138.1.fa.gz")
saveRDS(tax, here("data", "pilot_study", "tax.rds"))
saveRDS(taxsp, here("data", "pilot_study", "taxsp.rds"))

Save everything to phyloseq object

taxsp <- readRDS(here("data", "pilot_study", "taxsp.rds"))
ps <- phyloseq(sample_data(metadata),
               otu_table(st, taxa_are_rows = FALSE), 
               tax_table(taxsp))
# Add ASV sequences
seqs <- Biostrings::DNAStringSet(taxa_names(ps))
names(seqs) <- taxa_names(ps)
ps <- merge_phyloseq(ps, seqs)
taxa_names(ps) <- paste0("ASV", seq(ntaxa(ps)))

Construct the phylogenetic tree

set.seed(2024)
alignment <- AlignSeqs(refseq(ps), processors=NULL) 
tree <- Treeline(alignment, processors=NULL)
WriteDendrogram(tree, file=here('data','pilot_study','tree.nwk'))

add phylogenetic information to pyholseq objects

tree <- read_tree(here('data','pilot_study','tree.nwk'))
ps <- merge_phyloseq(ps, phy_tree(tree))
plot_tree(ps, method = 'treeonly')

Rarefied Data

The sequencing depth for each sample is different, we need to check and bring them to the same level.

# check distribution of sequencing depth
sample_sums(ps) %>% sort() %>% head()
##    020Petey   018Oakley    015Eames     019Jojo 016Fernando   022Archie 
##          38       16683       25228       25996       32452       36256

Here we see 020Petey has very low sequencing depth, we will remove the pair and rarefy the data to 16683.

sample_data(ps)['020Petey', 'Household']
##          Household
## 020Petey        10
ps <- subset_samples(ps, Household != '10')
# make read depth even
ps.rarefied <- rarefy_even_depth(ps, sample.size = 16683)
